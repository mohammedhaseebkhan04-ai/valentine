<script>
function playMusic() {
  document.getElementById("music").src += "&autoplay=1";
}

function sayYes() {
  document.getElementById("card").style.display = "none";
  document.getElementById("yes").style.display = "block";
  playMusic();
  startTyping();
}

function startTyping() {
  const text = "I love you Anitha, my forever happy place ‚ù§Ô∏è";
  let i = 0;
  const el = document.getElementById("typing");
  const interval = setInterval(() => {
    el.textContent += text.charAt(i);
    i++;
    if (i >= text.length) clearInterval(interval);
  }, 100);
}

function updateCountdown() {
  const now = new Date();
  let year = now.getFullYear();
  let v = new Date(`February 14, ${year} 00:00:00`);
  if (now > v) v = new Date(`February 14, ${year+1} 00:00:00`);
  const d = v - now;
  const days = Math.floor(d / (1000*60*60*24));
  const h = Math.floor((d/(1000*60*60))%24);
  const m = Math.floor((d/(1000*60))%60);
  const s = Math.floor((d/1000)%60);
  document.getElementById("countdown").innerHTML =
    `‚è≥ ${days} days ${h}h ${m}m ${s}s until our Valentine`;
}
setInterval(updateCountdown, 1000);
updateCountdown();
